# CMake files for libsrcml examples

# Any more recent cmake can be specified
cmake_minimum_required(VERSION 3.10)

project(srcML-Examples)

# Produces the target LibsrcML::LibsrcML
# Can also specify components. The default component is shared
#   find_package(libsrcml COMPONENT shared REQUIRED)
#   find_package(libsrcml COMPONENT static REQUIRED)
find_package(libsrcml REQUIRED)

# Add each executable as a separate program
# For each target, add the LibsrcML::LibsrcML library

add_executable(srcml_xslt srcml_xslt.c)
target_link_libraries(srcml_xslt PRIVATE LibsrcML::LibsrcML)
